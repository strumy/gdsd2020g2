CREATE TABLE `MARKET_TABLE` (
  `post_id` int(11) NOT NULL AUTO_INCREMENT,
  `title` varchar(255) NOT NULL,
  `author_id` int(11) NOT NULL,
  `date_inserted` timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  `market_category` varchar(40) NOT NULL,
  `Price` decimal(9,2) NOT NULL,
  `Status` varchar(100) NOT NULL,
  `Sold_to` int(11) DEFAULT NULL,
  `text_body` text,
  `picture` blob,
  PRIMARY KEY (`post_id`),
  KEY `author_id` (`author_id`),
  CONSTRAINT `MARKET_TABLE_ibfk_1` FOREIGN KEY (`author_id`) REFERENCES `users` (`id`) ON DELETE CASCADE
) ENGINE=InnoDB AUTO_INCREMENT=9 DEFAULT CHARSET=latin1
